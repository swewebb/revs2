# if

--

```php []
<?php

$a = "Hej";

if ($a == "hej") {
  echo "<h1>#1 " . $a . "</h1>";
}

if ($a == "Hej") {
  echo "<h1>#2 " . $a . "</h1>";
}
```

# if...else

--

```php []
<?php

$a = "Hej";

if ($a == "hej") {
  $b = "tjena teknisten!";
} else {
  $b = "Elsa är här!";
}

echo "<h1>" . $b . "</h1>";

```

--

![bild](bilder/php-villkor-if-else.png)

## Alternativa lösningar #1

```php []
<?php

$a = "Hej";
$kontroll = "hej";

if ($a == $kontroll) {
  $b = "tjena teknisten!";
}

if ($a != $kontroll) {
  $b = "Elsa är här!";
}

echo "<h1>" . $b . "</h1>";
```

# switch

--

```php []
<?php

$color = "blue";

switch ($color) {
  case "red":
    $result = "#f00";
    break;
  case "green":
    $result = "#0f0";
    break;
  case "blue":
    $result = "#00f";
    break;
  default:
    $result = "#b0b";
}

echo "<p>Färg: $result</p>";
```

# match

--

```php []
<?php

$color = "green";

$result = match ($color) {
  "red" => "#f00",
  "green" => "#0f0",
  "blue" => "#00f",
  default => "#b0b",
};

echo "<p>Färg: $result</p>";
```
